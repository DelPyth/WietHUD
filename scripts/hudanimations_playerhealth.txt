//Health Bonus Pulse
event HudHealthBonusPulse
{
	Animate	PlayerStatusHealthBonusImage 	Alpha		"150"		Linear 0.0 0.2
	Animate	PlayerStatusHealthBonusImage 	Alpha		"0"		Linear 0.2 0.4

	Animate PlayerHealthTextValue		FgColor		"HealthOverhealColor"	Linear 0.0 0.0
	Animate PlayerStatusHealthValue		FgColor		"HealthOverhealColor"	Linear 0.0 0.0
	Animate	PlayerStatusFreezeHealthValue 	FgColor		"HealthOverhealColor"	Linear 0.0 0.0
	Animate PlayerStatusSpecHealthValue	FgColor		"HealthOverhealColor"	Linear 0.0 0.0

	RunEvent HudHealthBonusPulseLoop	0.4
}

event HudHealthBonusPulseStop
{
	StopEvent HudHealthBonusPulse 0.0
	StopEvent HudHealthBonusPulseLoop 0.0

	Animate	PlayerHealthTextValue 		FgColor		"TanLight"		Linear 0.0 0.0
	Animate	PlayerStatusHealthValue 	FgColor		"HealthNormalColor"	Linear 0.0 0.0
	Animate	PlayerStatusFreezeHealthValue 	FgColor		"TanDark"		Linear 0.0 0.0
	Animate PlayerStatusSpecHealthValue	FgColor		"TanLight"		Linear 0.0 0.0
}
